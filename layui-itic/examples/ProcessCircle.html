基本使用参数配置

          基于:
          layui-processCircle.css; 
          jquery-1.9.1.min.js
          ProcessCircle.js
          调用：
          circle({
            el: $('#circle'), //需要设置进度环的元素
            value: 25,   //表示环形的百分比
            background: '#fff',  // 表示外部圆的颜色
            color: '#f0f',   //通过改变参数来改变圆环的颜色
            width: [120, 90],  //[外圆直径，内圆直径]通过改变来改变圆环的宽度
            deg: 90, // 表示环形的起始位置，起始位置为角度
          })
